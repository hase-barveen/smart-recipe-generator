  const resultsDiv = document.getElementById('results');
  const searchInput = document.getElementById('ingredient');
  const searchBtn = document.getElementById('searchBtn');
  const showAllBtn = document.getElementById('showAllBtn');

  function renderCards(list, target){
    target.innerHTML = '';
    if(!list || list.length===0){
      target.innerHTML = `<p style="padding:10px;color:#666">No recipes found.</p>`;
      return;
    }
    list.forEach(r => {
      const d = document.createElement('div');
      d.className = 'card';
      d.innerHTML = `
        <img src="${r.image}" alt="${r.name}">
        <div class="card-body">
          <h3>${r.name}</h3>
          <p>${r.description}</p>
          <div class="meta small" style="margin-top:8px">
            ${r.ingredients.map(i=>`<span class="chip">${i}</span>`).join('')}
          </div>
        </div>
      `;
      target.appendChild(d);
    });
  }

  resultsDiv.innerHTML = '<p class="small" style="padding:10px;color:#666">Search for an ingredient or recipe to see results here.</p>';

  function findRecipes(){
    const q = (searchInput.value || '').trim().toLowerCase();
    if(!q){
      resultsDiv.innerHTML = '<p class="small" style="padding:10px;color:#666">Please type something to search (e.g. "idli", "rice", "chicken").</p>';
      return;
    }
    const filtered = recipes.filter(r =>
      r.name.toLowerCase().includes(q) ||
      r.ingredients.some(ing => ing.toLowerCase().includes(q))
    );
    renderCards(filtered, resultsDiv);
  }

  searchBtn.addEventListener('click', findRecipes);
  searchInput.addEventListener('keydown', e => { if(e.key==='Enter') findRecipes(); });
  showAllBtn.addEventListener('click', ()=>{ renderCards(recipes, resultsDiv); });
